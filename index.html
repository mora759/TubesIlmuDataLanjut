    <script>

        // ==========================================

        // 1. LOGIKA REGRESI LINEAR

        // ==========================================

        function prediksiRegresi() {

            const umur = parseFloat(document.getElementById('inputUmur').value);

            const bmi = parseFloat(document.getElementById('inputBmi').value);


            if (isNaN(umur) || isNaN(bmi)) {

                alert("Harap masukkan angka yang valid untuk Umur dan BMI!");

                return;

            }


            // PENTING: Ganti angka di bawah ini dengan hasil cetakan (print) dari Google Colab Cell 5 milikmu!

            const intercept = -6317.02; // Ganti dengan angka intercept-mu

            const bobotUmur = 243.68;   // Ganti dengan angka bobot umur-mu

            const bobotBmi = 345.21;    // Ganti dengan angka bobot BMI-mu


            // Rumus regresi linear berganda

            const prediksi = intercept + (bobotUmur * umur) + (bobotBmi * bmi);


            const divHasil = document.getElementById('hasilRegresi');

            divHasil.classList.remove('hidden');

            divHasil.innerHTML = `Estimasi Biaya: <br> <span class="text-2xl">$${prediksi.toFixed(2)}</span>`;

        }


        // ==========================================

        // 2. LOGIKA KLASIFIKASI GAMBAR (CNN)

        // ==========================================

        let modelCNN;

        // Daftar nama kelas sesuai dataset (Dataset bunga: 5 kelas)

        const classNames = ['Daisy (Bunga Aster)', 'Dandelion', 'Roses (Mawar)', 'Sunflowers (Matahari)', 'Tulips'];


        // Fungsi memuat model dari folder 'tfjs_model'

        async function loadModel() {

            try {

                // Asumsinya folder tfjs_model berada di tempat (direktori) yang sama dengan index.html ini

                modelCNN = await tf.loadLayersModel('./tfjs_model/model.json');

                document.getElementById('statusModel').innerText = "✅ Model Gambar Siap Digunakan!";

                document.getElementById('statusModel').className = "text-xs font-bold text-green-600 mb-4";

            } catch (error) {

                console.error(error);

                document.getElementById('statusModel').innerText = "❌ Gagal memuat model. Pastikan folder tfjs_model sudah di-upload ke GitHub!";

                document.getElementById('statusModel').className = "text-xs font-bold text-red-600 mb-4";

            }

        }

        

        // Memuat model segera setelah halaman web terbuka

        loadModel();


        // Fitur untuk preview gambar yang diupload

        const inputGambar = document.getElementById('inputGambar');

        const previewGambar = document.getElementById('previewGambar');

        const placeholderText = document.getElementById('placeholderText');


        inputGambar.addEventListener('change', (event) => {

            const file = event.target.files[0];

            if (file) {

                const reader = new FileReader();

                reader.onload = (e) => {

                    previewGambar.src = e.target.result;

                    previewGambar.classList.remove('hidden');

                    placeholderText.classList.add('hidden');

                }

                reader.readAsDataURL(file);

            }

        });


        // Fungsi untuk melakukan prediksi pada gambar

        async function prediksiGambar() {

            if (!modelCNN) {

                alert("Model belum selesai dimuat. Tunggu sebentar.");

                return;

            }

            if (previewGambar.classList.contains('hidden')) {

                alert("Harap pilih dan masukkan gambar terlebih dahulu.");

                return;

            }


            const divHasil = document.getElementById('hasilKlasifikasi');

            divHasil.classList.remove('hidden');

            divHasil.innerText = "Memproses...";


            try {

                // Preprocessing citra agar sesuai dengan input model (128x128 pixel, rescale 1./255)

                const tensor = tf.browser.fromPixels(previewGambar)

                    .resizeNearestNeighbor([128, 128])

                    .toFloat()

                    .div(tf.scalar(255.0))

                    .expandDims();


                // Lakukan Prediksi

                const prediksi = await modelCNN.predict(tensor).data();

                

                // Cari probabilitas tertinggi

                const indeksTerbesar = prediksi.indexOf(Math.max(...prediksi));

                const persentase = (prediksi[indeksTerbesar] * 100).toFixed(2);

                const hasilKelas = classNames[indeksTerbesar];


                divHasil.innerHTML = `Terdeteksi: <br> <span class="text-2xl">${hasilKelas}</span> <br> <span class="text-sm font-normal text-gray-600">Akurasi deteksi: ${persentase}%</span>`;

            } catch (error) {

                console.error(error);

                divHasil.innerHTML = "<span class='text-red-500'>Terjadi kesalahan saat memproses gambar.</span>";

            }

        }

    </script>
